    <h1>Will This Get Me Sued?</h1>
    <p>Paste your text (tweet, email, post) to check for legal risks:</p>
    <textarea id="input" placeholder="e.g., 'This company scams people...'"></textarea>
    <button onclick="checkText()">Check Now</button>
    <div id="result"></div>

    <script>
        function checkText() {
            const text = document.getElementById("input").value.toLowerCase();
            const resultDiv = document.getElementById("result");

            // Risiko-Schlüsselwörter (Englisch, weil Rechtsbegriffe international ähnlich sind)
            const defamationWords = ["scam", "fraud", "criminal", "thief", "liar"];
            const copyrightWords = ["free download", "torrent", "watch online", "pdf"];
            const threatWords = ["sue you", "ruin you", "pay me", "destroy"];

            // Risiko prüfen
            let risk = "";
            if (defamationWords.some(word => text.includes(word))) {
                risk = "HIGH RISK: Potential defamation (false accusations can lead to lawsuits).";
            } else if (copyrightWords.some(word => text.includes(word))) {
                risk = "MEDIUM RISK: Copyright violation (sharing pirated content is illegal).";
            } else if (threatWords.some(word => text.includes(word))) {
                risk = "HIGH RISK: Threats can result in legal action.";
            } else {
                risk = "LOW RISK: No obvious legal threats detected.";
            }

            // Ergebnis anzeigen
            if (risk.includes("HIGH")) {
                resultDiv.className = "danger";
                resultDiv.innerHTML = `⚠️ ${risk}`;
            } else {
                resultDiv.className = "safe";
                resultDiv.innerHTML = `✅ ${risk}`;
            }
        }
    </script>
